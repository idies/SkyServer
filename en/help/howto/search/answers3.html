    select
        galaxy.objID, galaxy.u, galaxy.g,
        galaxy.r, galaxy.i, galaxy.z, specObj.z,
        plateX.plate, plateX.mjd, specObj.fiberID
    from
        galaxy, specObj, plateX
    where
        galaxy.objID=specObj.bestObjID 
        AND specObj.plateID=plateX.plateID 
        AND specObj.class='galaxy' 
        AND galaxy.g BETWEEN 17 and 17.4 
        AND specObj.z < 0.05
